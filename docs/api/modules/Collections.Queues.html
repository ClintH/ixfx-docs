  • queue<V>(opts?: QueueOpts, ...startingItems: readonly V[]): Queue<V>

  • Returns an immutable queue

    q = q.enqueue(`a`, `b`);   // Add two strings

    q = q.dequeue();           // q now just consists of `b`
